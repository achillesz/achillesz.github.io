arguments 

```js
    
    function mixArgs(first, second) {
       "use strict";
       console.log(first === arguments[0])
       console.log(second === arguments[1])
    
       first = 3;
       second = 4;
       console.log(first === arguments[0])
       console.log(second === arguments[1])
    }
    mixArgs(1, 2)
    true
    true
    false
    false
     
```
